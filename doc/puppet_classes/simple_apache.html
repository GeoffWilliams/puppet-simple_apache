class simple_apache(
  String $apache_name,
  String $service_name,
  String $conf_dir,
  String $confd_dir,
  String $mod_dir,
  String $vhost_dir,
  Boolean $manage_package = true,
  Boolean $manage_service = true,
) {

  if $manage_package {
    package { $apache_name:
      ensure => present,
    }
  }

  if $manage_service {
    service { $service_name:
      ensure => running,
      enable => true,
    }
  }

  file { $vhost_dir:
    ensure => directory,
  }


}
